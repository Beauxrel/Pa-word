; AddTwo.asm - adds two 32-bit integers.
; Chapter 3 example

.386
.model flat, stdcall
.stack 4096
ExitProcess PROTO, dwExitCode:DWORD
include Irvine32.inc

.data
;==============================================================================
input_string_msg			db "Enter a password: ", 0
input_function_msg			db 'Enter a password: ', 0
no_special_char				db 'Your password is lacking a special character',0Ah, 0Dh, 0
no_upper_char				db 'Your password is lacking an uppercase character',0
pass_short					db 'Password too short!',0Ah, 0Dh, 0
pass_long					db 'Password too long!',0Ah, 0Dh, 0
no_Upper_Msg				db 'Pasword must have a uppercase character',0Ah, 0Dh, 0
no_Lower_Msg				db 'Pasword must have a lowercase character',0Ah, 0Dh, 0
no_Number_Msg				db 'Pasword must have a number',0Ah, 0Dh, 0
no_Special_Msg				db 'Pasword must have a special character',0Ah, 0Dh, 0
Test_msg					db 'Good job!',0Ah, 0Dh, 0
special_chars				db "?@*()+{}&â€™_^[]\|", 0	
eek_msg						db "Too many passwords", 0
Redo_msg					db 'Password is identical. Please <enter> new password', 0
Retry						db 'Try again? (y/n) ', 0

;==============================================================================
gold_user_password				db 24 dup (0)
user_password					db 24 dup (0)
old_user_password				db 24 dup (0)
old_old_user_password			db 24 dup (0)
old_old_old_user_password		db 24 dup (0)
old_old_old_old_user_password	db 24 dup (0)
password_length					db		?
old_password_length				dd		?
old_old_password_length			dd		?
old_old_old_password_length		dd		?
old_old_old_old_password_length	dd		?
count							dd		?
;==============================================================================

.code
main PROC

	mov dl, 4
main_outer_loop:
	push dx
	xor edx, edx
	

String_Input:
	mov edx, offset input_string_msg
	call WriteString
	mov ecx, 24
	mov esi, 0
	;mov edx, offset user_password
	;mov password_length, al
String_Loop:
	xor eax, eax
	call readchar
	mov [user_password + esi], al
	inc esi
	cmp al, 0Dh
	je password_check
	mov al, '*'
	call writechar
	dec cx
	cmp cx, 0
	jz password_check
	jmp String_Loop
	;jmp password_check
;==================yes/no================
yes_no:
call readchar
call crlf
cmp al, 79h
je String_Input
cmp al, 6Eh
je endgame

;==============ERRORS====================
p2s_error:
	mov edx, offset pass_short
	call writestring
	mov edx, offset Retry
	call writestring
	jmp yes_no
p2l_error:
	mov edx, offset pass_long
	call writestring
	mov edx, offset Retry
	call writestring
	jmp yes_no
no_number:
	mov edx, offset no_Number_Msg
	call writestring
	mov edx, offset Retry
	call writestring
	jmp yes_no
no_Upper:
	mov edx, offset no_Upper_Msg
	call writestring
	mov edx, offset Retry
	call writestring
	jmp yes_no
no_Lower:
	mov edx, offset no_Lower_Msg
	call writestring
	mov edx, offset Retry
	call writestring
	jmp yes_no
no_special:
	mov edx, offset no_Special_Msg
	call writestring
	mov edx, offset Retry
	call writestring
	jmp yes_no
eek:
	mov edx, offset eek_msg
	call writestring
	jmp String_Input
;==============Length======================
password_check:
;password too short / too long
	call crlf
	dec esi
	;push esi
	;mov [password_length], esi
	;mov ah, [user_password]
	cmp esi, 0Ah
	jl p2s_error
	cmp esi, 16h
	jg p2l_error

;==============Number================
number:
	;mov cl, esi
	mov ecx, esi
	mov edx, offset user_password
number_Loop:
	mov ebx, [edx]
	cmp bl, 30h
	jb increment_upper
	cmp bl, 39h
	jbe upper
	cmp cl, 0
	jz no_Number
	jmp increment_Number
increment_Number:
	inc edx
	dec cl
	cmp cl, 0
	jz no_Number
	jmp number_Loop
;==============Upper Letter================
upper:
	mov ecx, esi
	mov edx, offset user_password
upper_Loop:
	mov ebx, [edx * 1]
	cmp bl, 'A'
	jb increment_upper
	cmp bl, 'Z'
	jbe lower
	cmp cl, 0
	jz no_Upper
	jmp increment_Upper
increment_upper:
	inc edx
	dec cl
	cmp cl, 0
	jz no_Upper
	jmp upper_Loop
;=============Lower Letter================
lower:
	mov ecx, esi
	mov edx, offset user_password
lower_Loop:
	mov ebx, [edx * 1]
	cmp bl, 'a'
	jb increment_Lower
	cmp bl, 'z'
	jbe special
	cmp cl, 0
	jz no_Lower
	jmp increment_Lower
increment_Lower:
	inc edx
	dec cl
	cmp cl, 0
	jz no_Lower
	jmp lower_Loop
;==============Special Character===============
special:
	mov edi, esi
	mov ecx, esi
	mov edx, offset user_password
	xor esi, esi

outer_Loop:
	push cx
	mov ebx, [edx * 1]
	mov esi, offset special_chars
	mov cl, 16
inside_Loop:
	mov al, BYTE PTR [esi]
	cmp bl, al
	je copy
	inc esi
	dec cl
	jcxz inside_Loop_Done
	jmp inside_Loop
inside_Loop_Done:
	pop cx
	dec cx
	inc edx
	cmp cx, 0
	jz no_special
	jmp outer_Loop

;==============Copy Password====================
copy:
pop cx
xor edx, edx
pop dx
mov ecx, edi
push cx
xor edi, edi
cmp dl, 3
je copy_to_Old_Old
cmp dl, 2
je copy_to_Old_Old_Old
cmp dl, 1
je copy_to_Old_Old_Old_Old

copy_to_Old:
    mov esi, offset user_password
    mov edi, offset old_old_old_user_password
    ;mov cl, password_length
copying_Loop_1:
    mov ebx, [esi * 1]
    mov [edi], bl
    inc esi
    inc edi
    dec cl
	cmp cl, 0
    jz copy_done
    jmp Copying_Loop_1

copy_to_Old_Old:
    mov esi, offset user_password
    mov edi, offset old_old_user_password
    ;mov cl, password_length
copying_Loop_2:
    mov ebx, [esi * 1]
    mov [edi], bl
    inc esi
    inc edi
    dec cl
	cmp cl, 0
    jz identical
    jmp Copying_Loop_2

copy_to_Old_Old_Old:
    mov esi, offset user_password
    mov edi, offset old_user_password
    ;mov cl, password_length
copying_Loop_3:
    mov ebx, [esi * 1]
    mov [edi], bl
    inc esi
    inc edi
    dec cl
	cmp cl, 0
    jz identical
    jmp Copying_Loop_3

copy_to_Old_Old_Old_Old:
    mov esi, offset user_password
    mov edi, offset gold_user_password
    ;mov cl, password_length
copying_Loop_4:
    mov ebx, [esi * 1]
    mov [edi], bl
    inc esi
    inc edi
    dec cl
	cmp cl, 0
    jz identical
    jmp Copying_Loop_4

copy_done:
dec dx
cmp dl, 0
jz endgame
jmp main_outer_loop
;==============Identical Password===============
identical:
	pop cx
	mov count, ecx
	push dx
	mov edx, 0
	;mov ecx, edi
	xor eax, eax
	xor ebx, ebx
	mov edi, offset old_old_old_user_password
	mov esi, offset user_password
	;mov cl, password_length
identical_loop:
	mov ebx, [esi]
	mov eax, [edi]
	cmp al, bl
	jne identical_2
	inc esi
	inc edi
	inc edx
	dec cl
	cmp cl, 0
	jz redo
	jmp identical_loop

identical_2:
	mov ecx, count
	pop dx
	cmp dl, 3
	je copy_done
	push dx
	mov edx, 0
	xor esi, esi
	xor eax, eax
	xor ebx, ebx
	mov edi, offset old_old_user_password
	mov esi, offset user_password
	;mov cl, password_length
identical_loop_2:
	mov ebx, [esi]
	mov eax, [edi]
	cmp al, bl
	jne identical_3
	inc esi
	inc edi
	inc edx
	dec cl
	cmp cl, 0
	jz redo
	jmp identical_loop_2

identical_3:
	mov ecx, count
	pop dx
	cmp dl, 2
	je copy_done
	push dx
	mov edx, 0
	xor edi, edi
	xor esi, esi
	xor eax, eax
	xor ebx, ebx
	mov edi, offset old_user_password
	mov esi, offset user_password
	;mov cl, password_length
identical_loop_3:
	mov ebx, [esi]
	mov eax, [edi]
	cmp al, bl
	jne identical_4
	inc esi
	inc edi
	inc edx
	dec cl
	jcxz redo
	jmp identical_loop_3

identical_4:
	mov ecx, count
	pop dx
	cmp dl, 1
	je copy_done
	push dx
	mov edx, 0
	je copy_done
	xor edi, edi
	xor esi, esi
	xor eax, eax
	xor ebx, ebx
	mov edi, offset gold_user_password
	mov esi, offset user_password
	;mov cl, password_length
identical_loop_4:
	mov ebx, [esi]
	mov eax, [edi]
	cmp al, bl
	jne copy_done
	inc esi
	inc edi
	inc edx
	dec cl
	jcxz redo
	jmp identical_loop_4

redo:
mov edx, offset Redo_msg
call writestring
call crlf
pop dx
jmp main_outer_loop

endgame:
mov edx, offset Test_msg
call writestring
invoke ExitProcess, 0 

;==============Identical Password===============
matching_pass:
	mov edx, offset Test_msg
	call writestring
	invoke ExitProcess, 0 

main ENDP

END main
